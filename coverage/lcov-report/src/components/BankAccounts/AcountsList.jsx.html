import React, {Component} from 'react';
import '../css/DashboardView.css';
 
// -- font-awesome ----------DashboardView---------------
import { FontAwesomeIcon } from '@fortawesome/react-fontawesome'
import { faDollarSign, faEdit, faBackspace } from '@fortawesome/free-solid-svg-icons'
//------------------------------------------
 
 
class AccountsList extends Component {
    render(props) {
 
        const accountsList = this.props.accountsArray.map((item, index) => {
            const balance = item.balance.toLocaleString('en-US', { style: 'currency', currency: 'USD' })
            return <tr key={'trAccountIndex' + index}>
                        <td><span 
                            id={'accountNameIndex' + index} 
                            key={'accountNameIndex' + index} 
                            onClick={this.props.clickHandler}
                            className={'accountListEntry' + (this.props.currentAccount === item.name ? ' active_account' : '')}
                            >{item.name} 
                        </span></td>
                        <td>&nbsp;&nbsp;</td>
                        <td><span 
                            id={'accountBlncIndex' + index} 
                            key={'accountBlncIndex' + index} 
                            onClick={this.props.clickHandler}
                            className={'accountListEntry' + (this.props.currentAccount === item.name ? ' active_account' : '')}
                            >{balance} 
                        </span></td>
                        <td><button
                            id={'depositAccountIndex' + index} 
                            key={'depositAccountIndex' + index} 
                            onClick={this.props.clickHandler}
                            >
                            <FontAwesomeIcon icon={faDollarSign} size="xs"></FontAwesomeIcon>
                        </button></td>
                        <td><button
                            id={'editAccountIndex' + index} 
                            key={'editAccountIndex' + index} 
                            onClick={this.props.clickHandler}
                            >
                            <FontAwesomeIcon icon={faEdit} size="xs"></FontAwesomeIcon>
                        </button></td>
                        <td><button
                            id={'delAccountIndex' + index} 
                            key={'delAccountIndex' + index} 
                            onClick={this.props.clickHandler}
                            className='severe'
                            >
                            <FontAwesomeIcon icon={faBackspace} size="xs"></FontAwesomeIcon>
                        </button></td>
                    </tr>
        })
  
        return (
            <div id="accountsListWrapper">
                <h1 className='accountslist'>
                    Accounts &nbsp;
                </h1>
 
                <table className='accountslist'>
                    <tbody>
                        <tr>
                            <td colSpan="5">
                                <button 
                                    id="addAccBtn" 
                                    onClick={this.props.clickHandler}
                                    >New account
                                </button>
                            </td>
                        </tr>
                        {accountsList}
                    </tbody>
                </table>
            </div>
        )
    }
}
export default AccountsList
