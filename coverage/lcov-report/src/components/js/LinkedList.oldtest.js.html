import { LinkedList } from './LinkedList'
 
 
describe('Linked List', () => {
    test('is instantiated without issue', () => {
 
        let newList = null
        newList = new LinkedList()
 
        expect(newList).not.toBeNull()
 
    })
})
 
describe('Linked List', () => {
    test('adding Nodes', () => {
 
        const testObj = { subject: 'banana', amount: 5 }
        const newList = new LinkedList()
        newList.print()
        newList.insert(testObj)
 
        console.log(`--> added first Node (head), length is: ${newList.length}`)
        newList.print()
 
        expect(newList.head).not.toBeNull()
        expect(newList.position).not.toBeNull()
        expect(newList.tail).not.toBeNull()
        expect(newList.length).toEqual(1)
        expect(newList.head.data.subject).toMatch(testObj.subject)
        expect(newList.head.data.amount).toEqual(testObj.amount)
 
        const testObj2 = { subject: 'apple', amount: 23 }
        newList.insert(testObj2)
 
        console.log(`--> added second Node (tail), length is: ${newList.length}`)
        newList.print()
 
        expect(newList.head.data.subject).toMatch(testObj.subject)
        expect(newList.head.data.amount).toEqual(testObj.amount)
        expect(newList.tail.data.subject).toMatch(testObj2.subject)
        expect(newList.tail.data.amount).toEqual(testObj2.amount)
 
        const testObj3 = { subject: 'kiwi', amount: 42 }
        newList.insert(testObj3)
 
        console.log(`--> added third Node (after 'position'), length is: ${newList.length}`)
        newList.print()
 
        expect(newList.head.data.subject).toMatch(testObj.subject)
        expect(newList.head.data.amount).toEqual(testObj.amount)
        expect(newList.head.next.data.subject).toMatch(testObj3.subject)
        expect(newList.head.next.data.amount).toEqual(testObj3.amount)
    })
})
